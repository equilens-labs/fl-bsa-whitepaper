
\section{Metric Definitions (Quick Reference)}
\begin{itemize}[leftmargin=*]
\item \textbf{AIR (disparity ratio)}: $p_g/p_{g_{\mathrm{ref}}}$, the selection-rate ratio for a protected group relative to a fixed reference group.
\item \textbf{SRG (approval-rate gap)}: $p_g - p_{g_{\mathrm{ref}}}$, the selection-rate difference for a protected group relative to a fixed reference group.
\item \textbf{ECE}: weighted absolute gap between empirical rate and predicted probability over calibration bins (expected calibration error).
\item \textbf{SAP}: pre-specified Statistical Analysis Plan describing estimands, thresholds, and inference methods for this audit scenario.
\end{itemize}
