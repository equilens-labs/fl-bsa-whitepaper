
\section{Statistical Analysis Plan (Summary)}
\begin{itemize}[leftmargin=*]
\item Significance $\alpha=0.05$; bootstrap resamples $B$ per SAP.
\item Wilson confidence intervals for proportions; BCa bootstrap CIs for AIR and equalised-odds gaps (\cref{eq:air,eq:eo-gap}); bootstrap CIs (BCa or percentile, per SAP) for ECE (\cref{eq:ece}) when evaluated.
\item Multiplicity: Benjamini--Hochberg at $q=0.10$.
\end{itemize}
